add_namespace = INT_italy

# March on Rome
country_event = {
	id = INT_italy.1
	title = INT_italy.1.t
	desc = INT_italy.1.d
	picture = GFX_report_event_march_on_rome

	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = ITA
		date > 1922.10.21
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		set_global_flag = march_on_rome
	}
	option = {
		name = INT_italy.1.a
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
			parties = {

				populist = { popularity = 0 }
				fascism = { popularity = 50 }
				paternal_autocrat = { popularity = 0 }
				social_conservative = { popularity = 20 }
				market_liberal = { popularity = 5 }
				social_liberal = { popularity = 5 }
				social_democrat = { popularity = 0 }
				leninist = { popularity = 5 }
				stalinist = { popularity = 0 }
				anarchist = { popularity = 10 }
			}
		}
		retire_country_leader = yes
		create_country_leader = {
			name = "Benito Mussolini"
			desc = "POLITICS_MUSSO_THE_DUSSO_DESC"
			picture = "Portrait_Italy_Benito_Mussolini.dds"
			expire = "1963.1.1"
			ideology = fascist
			traits = {
				dictator
			}
		}
	}
}
