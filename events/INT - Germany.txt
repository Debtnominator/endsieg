# Treaty of Versaille
country_event = {
	id = ww2_germany.13
	title = ww2_germany.13.t
	desc = ww2_germany.13.d
	picture = GFX_report_event_versaille

	is_triggered_only = yes

	option = {
		name = ww2_germany.13.a
		news_event = { hours = 6 id = news.20 }
		set_global_flag = versailles

		USA = { country_event = { id = usa.5 days = 1 } }
		WEI = {
			every_state = {
				limit = { is_core_of = WE1 }
				add_core_of = WEI
				remove_core_of = WE1
			}
			annex_country = { target = WE1 transfer_troops = no }
			change_tag_from = WE1
			42 = { set_demilitarized_zone = yes }
			51 = { set_demilitarized_zone = yes }
			764 = { set_demilitarized_zone = yes }
			add_political_power = -300
			add_manpower = 500000
			load_oob = "GER_1919"

			country_event = { id = ww2_germany.14 days = 150 }
			country_event = { id = ww2_germany.15 days = 10 }
			country_event = { id = ww2_germany.16 days = 3 }
			country_event = { id = ww2_germany.17 days = 1 }
			country_event = { id = ww2_germany.18 days = 625 }
            country_event = { id = ww2_germany.22 days = 1181 }
            country_event = { id = ww2_germany.28 days = 726 }
		}
		FR3 = {
			transfer_state = 28
		}
		POL = {
			transfer_state = 779
			if = {
				limit = { country_exists = RKP }
				annex_country = { target = RKP }
			}
			else = {
				transfer_state = 775
				transfer_state = 774
				transfer_state = 773
				transfer_state = 98
				transfer_state = 87
				transfer_state = 90
				transfer_state = 10
				transfer_state = 92
			}
		}
		BE2 = {
			transfer_state = 746
			transfer_state = 753
		}
	}
}
