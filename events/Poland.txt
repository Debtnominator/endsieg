# Ceasfire between Ukrops and Polaks
country_event = {
	id = poland.1
	title = poland.1.t
	desc = poland.1.d
	picture = GFX_report_event_ceasefire

	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		OR = {
			tag = UKR
			tag = POL
		}
		has_global_flag = regency_ends
		AND = {
			POL = {
				controls_state = 91
				controls_state = 93
			}
		}
		POL = { has_war_with = UKR }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = poland.1.a
		set_global_flag = polak_ukrop_ceasefire
		POL = {
			if = {
				limit = { country_exists = WUR }
				annex_country = { target = WUR transfer_troops = no }
			}
			transfer_state = 89
			transfer_state = 91
			transfer_state = 93
			change_tag_from = UKR
		}
		POL = { white_peace = UKR }
	}
}
