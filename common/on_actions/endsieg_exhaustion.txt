on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = {
					country_has_war_exhaustion = yes
				}
				set_variable = { war_exhaustion_variable = 0 }
				set_variable = { war_exhaustion_daily = 1 }
				set_variable = { peace_exhaustion_daily = -2 }
				set_variable = { war_exhaustion_max = 1825 }
			}
		}
	}

    on_daily = {
		effect = {
            if = {
				limit = { country_has_war_exhaustion = yes }

				if = {
					limit = { has_war = yes }
	                add_to_variable = { war_exhaustion_variable = war_exhaustion_daily }
				}

				if = {
					limit = {
						GE1 = { has_global_flag = welcome_event_1918 }
						NOT = { has_country_flag = war_exhaustion_1918_set }
					}
	                add_to_variable = { war_exhaustion_variable = 1332 }
					set_country_flag = war_exhaustion_1918_set
				}

				if = {
					limit = {
						has_global_flag = welcome_event_1941
						not = { has_country_flag = war_exhaustion_1941_set }
					}
	                add_to_variable = { war_exhaustion_variable = 250 }
					set_country_flag = war_exhaustion_1941_set
				}

				if = {
					limit = {
						has_global_flag = welcome_event_1943
						not = { has_country_flag = war_exhaustion_1943_set }
					}
	                add_to_variable = { war_exhaustion_variable = 500 }
					set_country_flag = war_exhaustion_1943_set
				}

				if = {
					limit = {
						has_global_flag = welcome_event_1944
						not = { has_country_flag = war_exhaustion_1944_set }
					}
	                add_to_variable = { war_exhaustion_variable = 750 }
					set_country_flag = war_exhaustion_1944_set
				}

				if = {
					limit = {
						has_global_flag = december_welcome_event_1944
						not = { has_country_flag = war_exhaustion_december_1944_set }
					}
	                add_to_variable = { war_exhaustion_variable = 1000 }
					set_country_flag = war_exhaustion_1943_set
				}

				if = {
					limit = {
						has_global_flag = welcome_event_1945
						not = { has_country_flag = war_exhaustion_1945_set }
					}
	                add_to_variable = { war_exhaustion_variable = 1250 }
					set_country_flag = war_exhaustion_1945_set
				}

				if = {
					limit = { has_war = no }
	                add_to_variable = { war_exhaustion_variable = peace_exhaustion_daily }
				}

				if = {
					limit = {
						check_variable = { war_exhaustion_variable < 365 }
						OR = {
							has_idea = slight_war_exhaustion_gui
							has_idea = slight_war_exhaustion_gui
							has_idea = slight_war_exhaustion_gui
						}
					}
					remove_ideas = slight_war_exhaustion_gui
				}


				if = {
					limit = {
						check_variable = { war_exhaustion_variable > 365 }
						check_variable = { war_exhaustion_variable < 730 }
						NOT = { has_idea = slight_war_exhaustion_gui }
					}
					if = {
						limit = { has_idea = generic_war_exhaustion_gui }
						remove_ideas = generic_war_exhaustion_gui
					}
					add_ideas = slight_war_exhaustion_gui
				}

				if = {
					limit = {
						check_variable = { war_exhaustion_variable > 730 }
						check_variable = { war_exhaustion_variable < 1095 }
						NOT = { has_idea = generic_war_exhaustion_gui }
					}
					if = {
						limit = { has_idea = slight_war_exhaustion_gui }
						remove_ideas = slight_war_exhaustion_gui
					}
					if = {
						limit = { has_idea = heavy_war_exhaustion_gui }
						remove_ideas = heavy_war_exhaustion_gui
					}
					add_ideas = generic_war_exhaustion_gui
				}

				if = {
					limit = {
						check_variable = { war_exhaustion_variable > 1095 }
						NOT = { has_idea = heavy_war_exhaustion_gui }
					}
					add_ideas = heavy_war_exhaustion_gui
					if = {
						limit = { has_idea = generic_war_exhaustion_gui }
						remove_ideas = generic_war_exhaustion_gui
					}
				}
                clamp_variable = { var = war_exhaustion_variable min = 0 max = war_exhaustion_max }
			}
		}
	}
}
